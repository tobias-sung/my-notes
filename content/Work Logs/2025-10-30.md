- Added a timeout for waiting for notification

```c title="UART Task" {20}
void vUARTTask(void* pvParameters) {
    irq_handler_t uart_irq_handler; 
    Proc uart_process;

    uart_inst_t* uart_inst= (uart_inst_t *) ((uintptr_t*)pvParameters)[0];
    uart_irq_handler = (irq_handler_t) ((uintptr_t*)pvParameters)[1];
    uart_process = (Proc) ((uintptr_t*)pvParameters)[2];
    
    if (uart_inst == uart0){
        irq_set_enabled(UART0_IRQ, true);
        uart_set_irqs_enabled(uart0, 1, 0); 
        irq_set_exclusive_handler(UART0_IRQ, uart_irq_handler);
    } else if (uart_inst == uart1){
        irq_set_enabled(UART1_IRQ, true);
        uart_set_irqs_enabled(uart1, 1, 0);
        irq_set_exclusive_handler(UART1_IRQ, uart_irq_handler);
    }

    uart_process();

}
```

### Modified GPIO Task to be more generic
Modified GPIO task in a similar way to UART task. Instead of a general GPIO interrupt handler that applies to all GPIO pins, different sets of GPIO pins can have their own distinct interrupt handlers and callback functions. 

The GPIO Task now accepts 3 input parameters: 
1. A bitmask indicating which GPIO pins should be monitored
2. A function pointer to an interrupt handler
3. A function pointer to a function that does certain work based on which GPIO pin received the interrupt

```c title="vGPIOTASK()"
void vGPIOTask(void *pvParameters){
    uint32_t gpio_mask; //Which GPIO pins this task will monitor
    irq_handler_t gpio_irq_handler;
    Proc gpio_process;
    
	//Extract input parameters
    gpio_mask = (uint32_t) ((uintptr_t*)pvParameters)[0];
    gpio_irq_handler = (irq_handler_t) ((uintptr_t*)pvParameters)[1];
    gpio_process = (Proc) ((uintptr_t*)pvParameters)[2];

    // Add the handler for the specified GPIOs
    gpio_add_raw_irq_handler_masked(gpio_mask, gpio_irq_handler);

    gpio_process();

}
```
